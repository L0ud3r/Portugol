        | SE e ENTAO com_list ';' SENAO com_list ';' FIMSE
        | SE e ENTAO com_list ';' fimse

e_list -> e | e_list ',' e

    | e '<' e | e leq e | e '>' e | e geq e | e '=' e | e dif e
